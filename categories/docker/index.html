命令

# 构建镜像（-t 给镜像起名，末尾的 . 表示当前目录）
docker build -t my-frontend .
# 运行容器(-d 后台运行，-p 端口映射，--name 容器名)4000-4100:4000-4100
docker run -d -p 8080:80 --name my-app my-frontend
# 查看正在运行的容器
docker ps
# 查看所有
docker ps -a
查看最近创建的容器
docker ps -l
# 删除单个容器
docker rm -f my-nginx
强行删除所有容器
docker rm -f $(docker ps -aq)
# 删除单个镜像
docker rmi my-frontend
# 查看有哪些镜像
docker images
# 重启名为 my-nginx 的容器
docker restart my-nginx
# 进入容器
docker exec -it 容器名称
# 启动名为 my-nginx 的容器
docker start my-nginx
# 直接进入容器运行mysql
docker exec -it mysql-container mysql -u root -p
# 查看容器里的日志
docker logs -f 容器ID

docker-compose批量创建镜像和容器

  1. 创建文件docker-compose.yml代码如下然后执行docker-compose up -d
services:
  # MySQL服务
  mysql:
    image: mysql:8.0
    container_name: mysql-container
    environment:
      # 时区上海
      TZ: Asia/Shanghai
      # root 密码
      MYSQL_ROOT_PASSWORD: root
      # 初始化数据库(后续的初始化sql会在这个库执行)
      MYSQL_DATABASE: ajmer-vue
    ports:
      - "3306:3306"
    volumes:
      - ./mysql_data:/var/lib/mysql
    command:
      # 将mysql8.0默认密码策略 修改为 原先 策略 (mysql8.0对其默认策略做了更改 会导致密码无法匹配)
      --default-authentication-plugin=mysql_native_password --character-set-server=utf8mb4 --collation-server=utf8mb4_general_ci --explicit_defaults_for_timestamp=true --lower_case_table_names=1
    restart: always
  # Nginx服务
  nginx:
    image: nginx:alpine
    container_name: nginx-container
    environment:
      # 时区上海
      TZ: Asia/Shanghai
    extra_hosts:
      # 让本地环境通过 127.0.0.1 访问宿主机上的 Java 应用
      # Java 容器环境会被 ajmer-admin 服务名覆盖
      - "ajmer-admin:host-gateway"
    ports:
      - "4000-4100:4000-4100"
    volumes:
      # 证书映射
      - ./nginx/cert:/etc/nginx/cert
      # 配置文件映射
      - ./nginx/conf:/etc/nginx/conf.d
      # 页面目录
      - ./nginx/html:/usr/share/nginx/html
      # 日志目录
      - ./nginx/log:/var/log/nginx
    privileged: true
    restart: unless-stopped
  # Redis服务
  redis:
    image: redis:alpine
    container_name: redis-container
    ports:
      - "6379:6379"
    environment:
      # 时区上海
      TZ: Asia/Shanghai
    privileged: true
    restart: always
    volumes:
      - ./redis-data:/data
  ajmer-admin:
    # 使用官方Java 17镜像
    image: eclipse-temurin:17-jdk
    # 容器名称
    container_name: back-container
    # 工作目录
    working_dir: /app
    # 端口映射：宿主机器端口:容器端口
    ports:
      - "8080:8080"
    # 复制JAR文件到容器
    volumes:
      - ./java:/app
    # 启动命令
    entrypoint: ["java", "-jar", "/app/ajmer-system.jar"]
    # 可选：设置重启策略
    restart: unless-stopped
    # 可选：设置环境变量
    environment:
      TZ: Asia/Shanghai
      DB_HOST: mysql
      REDIS_HOST: redis
      MINIO_HOST: minio
  #   # MinIO服务
  minio:
    image: minio/minio:RELEASE.2025-04-22T22-12-26Z
    container_name: minio-container
    ports:
      # API端口
      - "9000:9000"
      # 控制台端口
      - "9001:9001"
    environment:
      # 时区上海
      TZ: Asia/Shanghai
      # 管理后台用户名
      MINIO_ROOT_USER: admin
      # 管理后台密码(至少8位)
      MINIO_ROOT_PASSWORD: admin123456
    volumes:
      # 数据目录映射
      - ./minio/data:/data
      # 配置目录映射
      - ./minio/config:/root/.minio
    command: server /data --console-address ":9001"
    privileged: true
    restart: unless-stopped
  # kkFileView文件预览服务
  kkview:
    image: keking/kkfileview:4.1.0
    container_name: kkview-container
    ports:
      - "8012:8012"
    environment:
      # 时区上海
      TZ: Asia/Shanghai
    privileged: true
    restart: unless-stopped
# 定义数据卷
volumes:
  mysql_data: # 这里定义了mysql_data数据卷

linux上怎么安装docker

# 安装依赖
sudo apt update
sudo apt install -y ca-certificates curl
# 添加 Docker 官方 GPG Key
sudo install -m 0755 -d /etc/apt/keyrings
sudo curl -fsSL https://download.docker.com/linux/ubuntu/gpg -o /etc/apt/keyrings/docker.asc
sudo chmod a+r /etc/apt/keyrings/docker.asc
# 添加 Docker 官方软件源
sudo tee /etc/apt/sources.list.d/docker.sources <<EOF
Types: deb
URIs: https://download.docker.com/linux/ubuntu
Suites: $(. /etc/os-release && echo "${UBUNTU_CODENAME:-$VERSION_CODENAME}")
Components: stable
Architectures: $(dpkg --print-architecture)
Signed-By: /etc/apt/keyrings/docker.asc
EOF

# 然后更新
sudo apt update
# 安装 Docker
sudo apt install -y docker-ce docker-ce-cli containerd.io docker-buildx-plugin docker-compose-plugin

# 启动docker
sudo systemctl start docker
# 开启服务自动启动
sudo systemctl enable docker
# 查看docker状态和版本
sudo docker version
sudo docker ps
# 测试 Docker Compose
docker compose version

# 彻底停止docker
sudo systemctl stop docker.socket
sudo systemctl stop docker.service
sudo systemctl stop containerd

# 重启
sudo systemctl daemon-reload
sudo systemctl restart docker

监控服务方案2

services:
  # 监控服务方案2
  # 宿主机指标采集
  node-exporter:
    image: prom/node-exporter:latest
    container_name: node-exporter
    volumes:
      - /proc:/host/proc:ro
      - /sys:/host/sys:ro
      - /:/rootfs:ro
    command:
      - "--path.procfs=/host/proc"
      - "--path.sysfs=/host/sys"
      - "--collector.filesystem.mount-points-exclude=^/(sys|proc|dev|host|etc)($$|/)"
    restart: unless-stopped
  # MySQL 指标采集
  mysql-exporter:
    image: prom/mysqld-exporter:latest
    container_name: mysql-exporter
    command:
      - "--mysqld.username=root:root"
      - "--mysqld.address=mysql:3306"
    depends_on:
      - mysql
    restart: unless-stopped
  # Redis 指标采集
  redis-exporter:
    image: oliver006/redis_exporter:latest
    container_name: redis-exporter
    environment:
      REDIS_ADDR: "redis://redis:6379"
    depends_on:
      - redis
    restart: unless-stopped
  # Nginx 指标采集（需要在 nginx 配置里开启 stub_status，见下方说明）
  nginx-exporter:
    image: nginx/nginx-prometheus-exporter:latest
    container_name: nginx-exporter
    command:
      - "--nginx.scrape-uri=http://nginx:4002/stub_status"
    depends_on:
      - nginx
    restart: unless-stopped
  # Prometheus 核心
  prometheus:
    image: prom/prometheus:latest
    container_name: prometheus
    volumes:
      - ./prometheus:/etc/prometheus
      - prometheus_data:/prometheus
    command:
      - "--config.file=/etc/prometheus/prometheus.yml"
      - "--storage.tsdb.retention.time=30d"
    ports:
      - "9090:9090"
    restart: unless-stopped
  # Grafana 可视化
  grafana:
    image: grafana/grafana:latest
    container_name: grafana
    volumes:
      - grafana_data:/var/lib/grafana
    environment:
      TZ: Asia/Shanghai
      GF_SECURITY_ADMIN_PASSWORD: admin123 # 改成你自己的密码
      GF_USERS_ALLOW_SIGN_UP: "false"
    ports:
      - "3000:3000"
    depends_on:
      - prometheus
    restart: unless-stopped
# 定义数据卷
volumes:
  prometheus_data:
  grafana_data:
