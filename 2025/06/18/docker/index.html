安装地址

https://www.docker.com/

命令

# 构建镜像（-t 给镜像起名，末尾的 . 表示当前目录）
docker build -t my-frontend .
# 运行容器(-d 后台运行，-p 端口映射，--name 容器名)4000-4100:4000-4100
docker run -d -p 8080:80 --name my-app my-frontend
# 查看正在运行的容器
docker ps
# 查看所有
docker ps -a
查看最近创建的容器
docker ps -l
# 删除单个容器
docker rm -f my-nginx
强行删除所有容器
docker rm -f $(docker ps -aq)
# 删除单个镜像
docker rmi my-frontend
# 查看有哪些镜像
docker images
# 重启名为 my-nginx 的容器
docker restart my-nginx
# 进入容器
docker exec -it 容器名称
# 启动名为 my-nginx 的容器
docker start my-nginx
# 直接进入容器运行mysql
docker exec -it mysql-container mysql -u root -p

容器内的命令

# 查看所有已安装软件
apk info -v
# 查找特定软件
apk info | grep nginx

mysql 容器

docker run -d --name mysql-container -e MYSQL_ROOT_PASSWORD=my-secret-pw -e MYSQL_DATABASE=ry-vue -e MYSQL_USER=app_user -e MYSQL_PASSWORD=app_password -p 3306:3306 -v mysql_data:/var/lib/mysql --restart=always mysql:8.0 --character-set-server=utf8mb4 --collation-server=utf8mb4_unicode_ci

redis 容器

docker run -d --name redis-container -p 6379:6379 redis:alpine

GitLab服务

gitlab:
  image: gitlab/gitlab-ce:latest
  container_name: gitlab-container
  hostname: "gitlab.local"
  environment:
    GITLAB_OMNIBUS_CONFIG: |
      external_url 'http://localhost'
      gitlab_rails['gitlab_shell_ssh_port'] = 2222
      # 时间设置
      gitlab_rails['time_zone'] = 'Asia/Shanghai'
      # 电子邮件配置（可选）
      # gitlab_rails['smtp_enable'] = true
      # gitlab_rails['smtp_address'] = "smtp.example.com"
      # gitlab_rails['smtp_port'] = 587
    TZ: Asia/Shanghai
  ports:
    # HTTP端口
    - "8081:80"
    # HTTPS端口
    - "8443:443"
    # SSH端口（避免与主机22端口冲突）
    - "2222:22"
  volumes:
    # 配置文件
    - ./gitlab/config:/etc/gitlab
    # 数据目录
    - ./gitlab/data:/var/opt/gitlab
    # 日志目录
    - ./gitlab/logs:/var/log/gitlab
  privileged: true
  # GitLab需要较多资源
  deploy:
    resources:
      limits:
        memory: 4G
  restart: unless-stopped
  # GitLab启动较慢，给予充足的启动时间
  healthcheck:
    test: ["CMD", "/opt/gitlab/embedded/bin/gitlab-healthcheck", "fail"]
    interval: 30s
    timeout: 3m
    retries: 5
    start_period: 5m

docker-compose批量创建镜像和容器

  1. 创建文件docker-compose.yml代码如下然后执行docker-compose up -d
services:
  # MySQL服务
  mysql:
    image: mysql:8.0
    container_name: mysql-container
    environment:
      # 时区上海
      TZ: Asia/Shanghai
      # root 密码
      MYSQL_ROOT_PASSWORD: root
      # 初始化数据库(后续的初始化sql会在这个库执行)
      MYSQL_DATABASE: ajmer-vue
    ports:
      - "3306:3306"
    volumes:
      - ./mysql_data:/var/lib/mysql
    command:
      # 将mysql8.0默认密码策略 修改为 原先 策略 (mysql8.0对其默认策略做了更改 会导致密码无法匹配)
      --default-authentication-plugin=mysql_native_password --character-set-server=utf8mb4 --collation-server=utf8mb4_general_ci --explicit_defaults_for_timestamp=true --lower_case_table_names=1
    restart: always
  # Nginx服务
  nginx:
    image: nginx:alpine
    container_name: nginx-container
    environment:
      # 时区上海
      TZ: Asia/Shanghai
    extra_hosts:
      # 让本地环境通过 127.0.0.1 访问宿主机上的 Java 应用
      # Java 容器环境会被 ajmer-admin 服务名覆盖
      - "ajmer-admin:host-gateway"
    ports:
      - "4000-4100:4000-4100"
    volumes:
      # 证书映射
      - ./nginx/cert:/etc/nginx/cert
      # 配置文件映射
      - ./nginx/conf:/etc/nginx/conf.d
      # 页面目录
      - ./nginx/html:/usr/share/nginx/html
      # 日志目录
      - ./nginx/log:/var/log/nginx
    privileged: true
    restart: unless-stopped
  # Redis服务
  redis:
    image: redis:alpine
    container_name: redis-container
    ports:
      - "6379:6379"
    environment:
      # 时区上海
      TZ: Asia/Shanghai
    privileged: true
    restart: always
    volumes:
      - ./redis-data:/data
  ajmer-admin:
    # 使用官方Java 17镜像
    image: eclipse-temurin:17-jdk
    # 容器名称
    container_name: back-container
    # 工作目录
    working_dir: /app
    # 端口映射：宿主机器端口:容器端口
    ports:
      - "8080:8080"
    # 复制JAR文件到容器
    volumes:
      - ./java:/app
    # 启动命令
    entrypoint: ["java", "-jar", "/app/ajmer-system.jar"]
    # 可选：设置重启策略
    restart: unless-stopped
    # 可选：设置环境变量
    environment:
      TZ: Asia/Shanghai
      DB_HOST: mysql
      REDIS_HOST: redis
      MINIO_HOST: minio
  #   # MinIO服务
  minio:
    image: minio/minio:RELEASE.2025-04-22T22-12-26Z
    container_name: minio-container
    ports:
      # API端口
      - "9000:9000"
      # 控制台端口
      - "9001:9001"
    environment:
      # 时区上海
      TZ: Asia/Shanghai
      # 管理后台用户名
      MINIO_ROOT_USER: admin
      # 管理后台密码(至少8位)
      MINIO_ROOT_PASSWORD: admin123456
    volumes:
      # 数据目录映射
      - ./minio/data:/data
      # 配置目录映射
      - ./minio/config:/root/.minio
    command: server /data --console-address ":9001"
    privileged: true
    restart: unless-stopped
  # kkFileView文件预览服务
  kkview:
    image: keking/kkfileview:4.1.0
    container_name: kkview-container
    ports:
      - "8012:8012"
    environment:
      # 时区上海
      TZ: Asia/Shanghai
    privileged: true
    restart: unless-stopped
# 定义数据卷
volumes:
  mysql_data: # 这里定义了mysql_data数据卷
