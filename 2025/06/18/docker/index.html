安装地址

https://www.docker.com/

命令

# 构建镜像（-t 给镜像起名，末尾的 . 表示当前目录）
docker build -t my-frontend .
# 运行容器（-d 后台运行，-p 端口映射，--name 容器名）4000-4100:4000-4100
docker run -d -p 8080:80 --name my-app my-frontend
# 查看正在运行的容器
docker ps
# 查看所有
docker ps -a
查看最近创建的容器
docker ps -l
# 删除单个容器
docker rm -f my-nginx
强行删除所有容器
docker rm -f $(docker ps -aq)
# 删除单个镜像
docker rmi my-frontend
# 查看有哪些镜像
docker images
# 重启名为 my-nginx 的容器
docker restart my-nginx
# 进入容器
docker exec -it 容器名称
# 启动名为 my-nginx 的容器
docker start my-nginx
# 直接进入容器运行mysql
docker exec -it mysql-container mysql -u root -p

容器内的命令

# 查看所有已安装软件
apk info -v
# 查找特定软件
apk info | grep nginx

docker-compose.yml

services:
  web:
    build: . # 使用当前目录的 Dockerfile 构建
    image: web:1.0 # 镜像名称
    container_name: nginx-container # 容器名称
    ports:
      - "4000-4100:4000-4100/tcp"
      - "80:80" # 实际 Web 端口
