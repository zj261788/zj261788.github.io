明道云插件开发

明道云 updateWorksheetRow 接口中，下拉类型字段（单选 type=11、多选 type=10）的 value 必须是 JSON 数组格式的选项 key，如 "[\"844f2012-1ee5-4d70-b08e-68da32678c6c\"]"，不能直接传显示文本。普通文本字段直接传字符串即可。

Why: 明道云 API 对不同控件类型的 value 格式有严格区分，下拉字段用选项 key 的 JSON 数组，文本字段用纯字符串。传错格式会导致 400 报错。

How to apply: 编辑表格单元格时，需通过 fieldMeta 判断字段类型。下拉字段用 el-select 渲染，选中后发送选项 key 并用 JSON.stringify([key]) 包装；本地状态则需把 key 反查回显示文本以保持表格显示正确。同时注意在组件间传递事件时，必须把 isOption 标志一路透传到底层 API 调用处，不能在中途丢失。

明道云插件开发必须使用 mdye 提供的 API，不要直接使用标准 fetch 或 axios

Why: 明道云插件运行在沙箱环境中，有内置的认证和请求封装。直接调用外部 API 会缺少必要的 token 和上下文信息。

/**
 * 格式化单元格值
 */
function formatCellValue(value, type, control = null) {
  if (value === null || value === undefined || value === "") {
    return "-";
  }
  try {
    switch (type) {
      case 15: // 日期
        return new Date(value).toLocaleDateString();
      case 16: // 时间
        return new Date(value).toLocaleString();
      case 3: // 数字
        return Number(value).toLocaleString();
      case 29: // 关联表
        if (value) {
          const arrayObjects = JSON.parse(value);
          // 空值处理：确保arrayObjects是数组
          if (Array.isArray(arrayObjects) && arrayObjects.length === 0) {
            return "";
          }
          if (!Array.isArray(arrayObjects)) {
            return value;
          }
          // 返回所有关联记录的名称，用逗号拼接
          return arrayObjects.map((item) => item.name || item.title || "-").join(", ");
        }
        break;
      case 11: // 单选
        if (value && control?.options) {
          const arrayObjects = JSON.parse(value);
          // 空值处理：确保arrayObjects是数组
          if (!Array.isArray(arrayObjects) || arrayObjects.length === 0) {
            return "-";
          }
          const optionObj = control.options.find((item) => item.key === arrayObjects[0]);
          return optionObj?.value || "-";
        }
        break;
      case 10: // 多选
        if (value && control?.options) {
          const arrayObjects = JSON.parse(value);
          // 空值处理：确保arrayObjects是数组
          if (!Array.isArray(arrayObjects) || arrayObjects.length === 0) {
            return "-";
          }
          const optionObjs = arrayObjects
            .map((id) => {
              return control.options.find((item) => item.key === id);
            })
            .filter(Boolean);
          return optionObjs.map((option) => option.value).join(", ");
        }
        break;
      case 26: // 成员
        if (value) {
          const arrayObjects = JSON.parse(value);
          // 空值处理：确保arrayObjects是数组
          if (!Array.isArray(arrayObjects) || arrayObjects.length === 0) {
            return "-";
          }
          // 返回所有成员姓名，用逗号拼接
          return arrayObjects.map((member) => member.fullname || member.accountId || "-").join(", ");
        }
        break;
      case 27: // 部门
        if (value) {
          const arrayObjects = JSON.parse(value);
          // 空值处理：确保arrayObjects是数组
          if (!Array.isArray(arrayObjects) || arrayObjects.length === 0) {
            return "-";
          }
          // 返回所有部门名称，用逗号拼接
          return arrayObjects.map((dept) => dept.departmentName || "-").join(", ");
        }
        break;
      case 24: // 地区
        if (value) {
          const obj = JSON.parse(value);
          if (!obj || obj.length === 0) {
            return "-";
          }
          return obj.name || "-";
        }
        break;
      case 48: // 组织角色
        if (value) {
          const arrayObjects = JSON.parse(value);
          // 空值处理：确保arrayObjects是数组
          if (!Array.isArray(arrayObjects) || arrayObjects.length === 0) {
            return "-";
          }
          // 返回所有组织角色名称，用逗号拼接
          return arrayObjects.map((role) => role.organizeName || "-").join(", ");
        }
        break;
      case 14: // 附件
        if (value) {
          const arrayObjects = JSON.parse(value);
          // 空值处理：确保arrayObjects是数组
          if (!Array.isArray(arrayObjects) || arrayObjects.length === 0) {
            return "-";
          }
          // 显示所有附件，用逗号分隔
          return arrayObjects.map((file) => `${file.originalFilename || ""}${file.ext || ""}`).join(", ");
        }
        break;
      case 42: // 签名
        if (value) {
          return "[签名图片]";
        }
        break;
      case 40: // 定位
        if (value) {
          const obj = JSON.parse(value);
          if (!obj || obj.length === 0) {
            return "-";
          }
          return obj.address || "-";
        }
        break;
      case 34: // 子表
        // 在主表格中显示为子表记录数
        if (value) {
          return `子表 (${value}条)`;
        }
        return "子表 (0条)";
      default:
        return String(value || "");
    }
  } catch (error) {
    console.warn("格式化单元格值失败:", error, "value:", value, "type:", type);
    return String(value || "");
  }
}

const handleAddRecord = (value) => {
  navGroupFilters = value.navGroupFilters;
  loadTableData();
};

  init();
  md_emitter.addListener("filters-update", handleFiltersUpdate);
  md_emitter.addListener("new-record", handleAddRecord);
});
